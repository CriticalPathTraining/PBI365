Sales Growth PM = 
  IF(
    ISFILTERED(Calendar[Month]) && 
    NOT(ISFILTERED(Calendar[Date]))
  ), 
  DIVIDE(  
    SUM(Sales[SalesAmount]) -
    CALCULATE(
      SUM(Sales[SalesAmount]),
      PREVIOUSMONTH(Calendar[Date])
    ), 
    CALCULATE(
      SUM(Sales[SalesAmount]),
      PREVIOUSMONTH(Calendar[Date])
    )
  ),
  BLANK()
)
